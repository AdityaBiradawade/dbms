1) CREATE DATABASE course;

2) use course

3) CREATE TABLE Department(
   dept_id INT PRIMARY KEY AUTO_INCREMENT,
   dept_name VARCHAR(50) );

4)  CREATE TABLE student(
    stud_id INT PRIMARY KEY,
    stud_name VARCHAR(50),
    gender CHAR(10),
    dob DATE,
    dept_id INT,
    email VARCHAR(50),
    CONSTRAINT fk_dept FOREIGN KEY (dept_id) REFERENCES Department(dept_id));

5) CREATE INDEX idx_student_name ON Student(stud_name);

6) CREATE VIEW v_student_details AS
   SELECT s.stud_id, s.stud_name, d.dept_name
   FROM Student s JOIN Department d ON s.dept_id = d.dept_id;
